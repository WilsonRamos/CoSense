165 * Transforms matrix to the row reduced echelon form (RREF)
166 */
167 static void
168 transformMatrixToRREF(ColMajorOrderMatrixXd & matrix, int nonPivotColumnIndices[], int matrixRank, int currentRow = 0, int currentColumn = 0)
169 {
170 int rowCount = matrix.rows();
171 int columnCount = matrix.cols();
172
173 if (currentRow >= rowCount)
174 {
175 /*
176 * The last columns may not have been checked for swaps
177 * and should be non-pivots.
178 */
179 int columnNumber = columnCount - 1;
180 int nonPivotIndex = columnCount - matrixRank - 1;
181 while (nonPivotColumnIndices[nonPivotIndex] == -1)
182 {
183 nonPivotColumnIndices[nonPivotIndex] = columnNumber;
184 columnNumber--;
185 nonPivotIndex--;
186 }
187
188 return;
189 }
190
191 /*
192 * Make the current entry non-zero
193 */
194 currentColumn = makeRREFSwapIfNecessary(matrix, nonPivotColumnIndices, matrixRank, currentRow, currentColumn);
195 if (currentColumn >= columnCount)
196 {
197 return;
198 }
199
200 /*
201 * Divide the current row through by the current entry
202 */
203 matrix.row(currentRow) /= matrix(currentRow, currentColumn);
204
205 /*
206 * Make all other entries in current column zero by subtracting
207 * multiples of the current row (which has a value of 1 at the
208 * current position).
209 */
210 for (int q = 0; q < rowCount; q++)
211 {
212 if (q != currentRow)
213 {
214 matrix.row(q) -= matrix.row(currentRow) * matrix(q, currentColumn);
215 }
216 }
217
218 /*
219 * Repeat the algorithm, starting at the next position
220 */
221 return transformMatrixToRREF(matrix, nonPivotColumnIndices, matrixRank, currentRow + 1, currentColumn + 1);

275 * For a dimensional matrix with n parameters (n columns) and rank r,
276 * (i.e., equivalently, with r pivot columns ==> r linearly-independent
277 * columns) there are choose(n, r) ways in which we can rearrange the
278 * columns of the dimensional matrix to yield a unique Pi group (see
279 * Harald Hanche-Olsen, 2004 and E. Buckingham, 1914).
280 *
281 * For each of these orderings of the r pivot columns, we generate
282 * all the possible n-bit words which have r '1's and we place the
283 * r pivot column indices at those positions.
284 *
285 * If we also wanted to account not just for placement of the pivot
286 * columns, but also for their permutations relative to each other,
287 * we could use Heap's algorithm (B. R. Heap, "Permutations by
288 * interchanges". The Computer Journal. 6 (3): 293–4) to find all the
289 * r! permutations of the r pivot column indices.

293 * uint64_t for the pivot position bitmap.
