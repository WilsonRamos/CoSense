165 * Transforms matrix to the row reduced echelon form (RREF)
166 */
167 static void
168 transformMatrixToRREF(ColMajorOrderMatrixXd & matrix, int nonPivotColumnIndices[], int matrixRank, int currentRow = 0, int currentColumn = 0)
169 {
170 int rowCount = matrix.rows();
171 int columnCount = matrix.cols();
172
173 if (currentRow >= rowCount)
174 {
175 /*
176 * The last columns may not have been checked for swaps
177 * and should be non-pivots.
178 */
179 int columnNumber = columnCount - 1;
180 int nonPivotIndex = columnCount - matrixRank - 1;
181 while (nonPivotColumnIndices[nonPivotIndex] == -1)
182 {
183 nonPivotColumnIndices[nonPivotIndex] = columnNumber;
184 columnNumber--;
185 nonPivotIndex--;
186 }
187
188 return;
189 }
190
191 /*
192 * Make the current entry non-zero
193 */
194 currentColumn = makeRREFSwapIfNecessary(matrix, nonPivotColumnIndices, matrixRank, currentRow, currentColumn);
195 if (currentColumn >= columnCount)
196 {
197 return;
198 }
199
200 /*
201 * Divide the current row through by the current entry
202 */
203 matrix.row(currentRow) /= matrix(currentRow, currentColumn);
204
205 /*
206 * Make all other entries in current column zero by subtracting
207 * multiples of the current row (which has a value of 1 at the
208 * current position).
209 */
210 for (int q = 0; q < rowCount; q++)
211 {
212 if (q != currentRow)
213 {
214 matrix.row(q) -= matrix.row(currentRow) * matrix(q, currentColumn);
215 }
216 }
217
218 /*
219 * Repeat the algorithm, starting at the next position
220 */
221 return transformMatrixToRREF(matrix, nonPivotColumnIndices, matrixRank, currentRow + 1, currentColumn + 1);

275 * For a dimensional matrix with n parameters (n columns) and rank r,
276 * (i.e., equivalently, with r pivot columns ==> r linearly-independent
277 * columns) there are choose(n, r) ways in which we can rearrange the
278 * columns of the dimensional matrix to yield a unique Pi group (see
279 * Harald Hanche-Olsen, 2004 and E. Buckingham, 1914).
280 *
281 * For each of these orderings of the r pivot columns, we generate
282 * all the possible n-bit words which have r '1's and we place the
283 * r pivot column indices at those positions.
284 *
285 * If we also wanted to account not just for placement of the pivot
286 * columns, but also for their permutations relative to each other,
287 * we could use Heap's algorithm (B. R. Heap, "Permutations by
288 * interchanges". The Computer Journal. 6 (3): 293–4) to find all the
289 * r! permutations of the r pivot column indices.

291 * Limitations:
292 * (1) We assume inputMatrix has less than 64 columns as we use a
293 * uint64_t for the pivot position bitmap.

96/** \internal Like malloc, but the returned pointer is guaranteed to be 16-byte aligned.
97 * Fast, but wastes 16 additional bytes of memory. Does not throw any exception.
98 */
99inline void* handmade_aligned_malloc(std::size_t size)
100{
101 void *original = std::malloc(size+EIGEN_DEFAULT_ALIGN_BYTES16);
102 if (original == 0) return 0;
103 void *aligned = reinterpret_cast<void*>((reinterpret_cast<std::size_t>(original) & ~(std::size_t(EIGEN_DEFAULT_ALIGN_BYTES16-1))) + EIGEN_DEFAULT_ALIGN_BYTES16);
104 *(reinterpret_cast<void**>(aligned) - 1) = original;
105 return aligned;
106}

108/** \internal Frees memory allocated with handmade_aligned_malloc */
109inline void handmade_aligned_free(void *ptr)
110{
111 if (ptr) std::free(*(reinterpret_cast<void**>(ptr) - 1));
112}

114/** \internal
115 * \brief Reallocates aligned memory.
116 * Since we know that our handmade version is based on std::malloc
117 * we can use std::realloc to implement efficient reallocation.
118 */
119inline void* handmade_aligned_realloc(void* ptr, std::size_t size, std::size_t = 0)
120{
121 if (ptr == 0) return handmade_aligned_malloc(size);
122 void *original = *(reinterpret_cast<void**>(ptr) - 1);
123 std::ptrdiff_t previous_offset = static_cast<char *>(ptr)-static_cast<char *>(original);
124 original = std::realloc(original,size+EIGEN_DEFAULT_ALIGN_BYTES16);
125 if (original == 0) return 0;
126 void *aligned = reinterpret_cast<void*>((reinterpret_cast<std::size_t>(original) & ~(std::size_t(EIGEN_DEFAULT_ALIGN_BYTES16-1))) + EIGEN_DEFAULT_ALIGN_BYTES16);
127 void *previous_aligned = static_cast<char *>(original)+previous_offset;
128 if(aligned!=previous_aligned)
129 std::memmove(aligned, previous_aligned, size);
130
131 *(reinterpret_cast<void**>(aligned) - 1) = original;
132 return aligned;
133}

458 * \param array the address of the start of the array
459 * \param size the size of the array
460 *
461 * \note If no element of the array is well aligned or the requested alignment is not a multiple of a scalar,
462 * the size of the array is returned. For example with SSE, the requested alignment is typically 16-bytes. If
463 * packet size for the given scalar type is 1, then everything is considered well-aligned.
464 *
465 * \note Otherwise, if the Alignment is larger that the scalar size, we rely on the assumptions that sizeof(Scalar) is a
466 * power of 2. On the other hand, we do not assume that the array address is a multiple of sizeof(Scalar), as that fails for
467 * example with Scalar=double on certain 32-bit platforms, see bug #79.
468 *
469 * There is also the variant first_aligned(const MatrixBase&) defined in DenseCoeffsBase.h.
470 * \sa first_default_aligned()
471 */
472template<int Alignment, typename Scalar, typename Index>
473EIGEN_DEVICE_FUNC inline Index first_aligned(const Scalar* array, Index size)
474{
475 const Index ScalarSize = sizeof(Scalar);
476 const Index AlignmentSize = Alignment / ScalarSize;
477 const Index AlignmentMask = AlignmentSize-1;
478
479 if(AlignmentSize<=1)
480 {
481 // Either the requested alignment if smaller than a scalar, or it exactly match a 1 scalar
482 // so that all elements of the array have the same alignment.
483 return 0;
484 }
485 else if( (UIntPtr(array) & (sizeof(Scalar)-1)) || (Alignment%ScalarSize)!=0)
486 {
487 // The array is not aligned to the size of a single scalar, or the requested alignment is not a multiple of the scalar size.
488 // Consequently, no element of the array is well aligned.
489 return size;
490 }
491 else
492 {
493 Index first = (AlignmentSize - (Index((UIntPtr(array)/sizeof(Scalar))) & AlignmentMask)) & AlignmentMask;
494 return (first < size) ? first : size;
495 }
496}

498/** \internal Returns the index of the first element of the array that is well aligned with respect the largest packet requirement.
499 * \sa first_aligned(Scalar*,Index) and first_default_aligned(DenseBase<Derived>) */
500template<typename Scalar, typename Index>
501EIGEN_DEVICE_FUNC inline Index first_default_aligned(const Scalar* array, Index size)
502{
503 typedef typename packet_traits<Scalar>::type DefaultPacketType;
504 return first_aligned<unpacket_traits<DefaultPacketType>::alignment>(array, size);
505}

507/** \internal Returns the smallest integer multiple of \a base and greater or equal to \a size
508 */
509template<typename Index>
510inline Index first_multiple(Index size, Index base)
511{
512 return ((size+base-1)/base)*base;

819/** \class aligned_allocator
820* \ingroup Core_Module
821*
822* \brief STL compatible allocator to use with types requiring a non standrad alignment.
823*
824* The memory is aligned as for dynamically aligned matrix/array types such as MatrixXd.
825* By default, it will thus provide at least 16 bytes alignment and more in following cases:
826* - 32 bytes alignment if AVX is enabled.
827* - 64 bytes alignment if AVX512 is enabled.
828*
829* This can be controlled using the \c EIGEN_MAX_ALIGN_BYTES macro as documented
830* \link TopicPreprocessorDirectivesPerformance there \endlink.
831*
832* Example:
833* \code
834* // Matrix4f requires 16 bytes alignment:
835* std::map< int, Matrix4f, std::less<int>,
836* aligned_allocator<std::pair<const int, Matrix4f> > > my_map_mat4;
837* // Vector3f does not require 16 bytes alignment, no need to use Eigen's allocator:
838* std::map< int, Vector3f > my_map_vec3;
839* \endcode
840*
841* \sa \blank \ref TopicStlContainers.
842*/
843template<class T>
844class aligned_allocator : public std::allocator<T>
845{
846public:
847 typedef std::size_t size_type;
848 typedef std::ptrdiff_t difference_type;
849 typedef T* pointer;
850 typedef const T* const_pointer;
851 typedef T& reference;
852 typedef const T& const_reference;
853 typedef T value_type;
854
855 template<class U>
856 struct rebind
857 {
858 typedef aligned_allocator<U> other;
859 };
860
861 aligned_allocator() : std::allocator<T>() {}
862
863 aligned_allocator(const aligned_allocator& other) : std::allocator<T>(other) {}
864
865 template<class U>
866 aligned_allocator(const aligned_allocator<U>& other) : std::allocator<T>(other) {}
867
868 ~aligned_allocator() {}
869
870 pointer allocate(size_type num, const void* /*hint*/ = 0)
871 {
872 internal::check_size_for_overflow<T>(num);
873 return static_cast<pointer>( internal::aligned_malloc(num * sizeof(T)) );
874 }
875
876 void deallocate(pointer p, size_type /*num*/)
877 {
878 internal::aligned_free(p);
879 }
880};

916inline void queryCacheSizes_intel_direct(int& l1, int& l2, int& l3)
917{
918 int abcd[4];
919 l1 = l2 = l3 = 0;
920 int cache_id = 0;
921 int cache_type = 0;
922 do {
923 abcd[0] = abcd[1] = abcd[2] = abcd[3] = 0;
924 EIGEN_CPUID(abcd,0x4,cache_id)__asm__ __volatile__ ("xchg{q}\t{%%}rbx, %q1; cpuid; xchg{q}\t{%%}rbx, %q1"
: "=a" (abcd[0]), "=&r" (abcd[1]), "=c" (abcd[2]), "=d" (
abcd[3]) : "0" (0x4), "2" (cache_id));;
925 cache_type = (abcd[0] & 0x0F) >> 0;
926 if(cache_type==1||cache_type==3) // data or unified cache
927 {
928 int cache_level = (abcd[0] & 0xE0) >> 5; // A[7:5]
929 int ways = (abcd[1] & 0xFFC00000) >> 22; // B[31:22]
930 int partitions = (abcd[1] & 0x003FF000) >> 12; // B[21:12]
931 int line_size = (abcd[1] & 0x00000FFF) >> 0; // B[11:0]
932 int sets = (abcd[2]); // C[31:0]
933
934 int cache_size = (ways+1) * (partitions+1) * (line_size+1) * (sets+1);
935
936 switch(cache_level)
937 {
938 case 1: l1 = cache_size; break;
939 case 2: l2 = cache_size; break;
940 case 3: l3 = cache_size; break;
941 default: break;
942 }
943 }
944 cache_id++;
945 } while(cache_type>0 && cache_id<16);
946}
